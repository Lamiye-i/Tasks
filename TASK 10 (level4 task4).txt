#include <iostream>
using namespace std;

void numbersWithDigitSum10() {
    for (int i = 100; i <= 999; i++) {
        int eded = i;
        int sum = 0;
        while (eded > 0) {
            sum += eded % 10;
            eded /= 10;
        }
        if (sum == 10) {
            cout << i << " ";
        }
    }
    cout << endl;
}

int main() {
    numbersWithDigitSum10();
    return 0;
}
