#include <iostream>
using namespace std;

int main() {
    int n, x, max, min; 
    long sum = 0;
    cin >> n;
    cin >> x; max = min = x; sum += x;
    for(int i = 2; i <= n; i++) {
        cin >> x;
        if(x > max) max = x;
        if(x < min) min = x;
        sum += x;
    }
    cout << max << " " << min << " " << (sum / n);
}
