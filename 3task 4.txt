#include <iostream>
using namespace std;

int main() {
    int n;
    long long faktorial = 1;

    cout << "Ededi daxil edin: ";
    cin >> n;

    for(int i = 1; i <= n; i++) {
        faktorial = faktorial * i;
    }

    cout << n << " ededinin faktoriali: " << faktorial;
    return 0;
}
