#include <iostream>
using namespace std;

int main() {
    double a, b;
    char op;

    cout << "Birinci ededi daxil edin: ";
    cin >> a;

    cout << "Operatoru daxil edin (+, -, *, /): ";
    cin >> op;

    cout << "Ikinci ededi daxil edin: ";
    cin >> b;

    switch(op) {
        case '+':
            cout << "Netice: " << a + b;
            break;

        case '-':
            cout << "Netice: " << a - b;
            break;

        case '*':
            cout << "Netice: " << a * b;
            break;

        case '/':
            if (b == 0)
                cout << "0-a bolmek olmaz!";
            else
                cout << "Netice: " << a / b;
            break;

        default:
            cout << "Duzgun operator daxil edin!";
    }

    return 0;
}
